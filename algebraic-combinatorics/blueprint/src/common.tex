% Shared macros for both web and print versions.

% Theorem-like environments tracked by the dependency graph.
% All share one counter and are numbered within chapters.
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}
\newtheorem{example}[theorem]{Example}

% Additional environments not in the dependency graph
\theoremstyle{remark}
\newtheorem{remark}[theorem]{Remark}
\newtheorem{convention}[theorem]{Convention}
\newtheorem{exercise}[theorem]{Exercise}

% Custom macros from the textbook
\newcommand{\QQ}{\mathbb{Q}}
\newcommand{\set}[1]{\left\{ #1 \right\}}
\newcommand{\abs}[1]{\left| #1 \right|}
\newcommand{\tup}[1]{\left( #1 \right)}
\newcommand{\ive}[1]{\left[ #1 \right]}
\newcommand{\floor}[1]{\left\lfloor #1 \right\rfloor}

% Convenience
\providecommand{\dbinom}[2]{\binom{#1}{#2}}
\providecommand{\dfrac}[2]{\frac{#1}{#2}}
\providecommand{\llbracket}{\ensuremath{[\![}}
\providecommand{\rrbracket}{\ensuremath{]\!]}}
\newcommand{\xpm}{x^{\pm}}
